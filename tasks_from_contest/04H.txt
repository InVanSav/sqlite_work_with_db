SELECT 
BEASTS.NAME AS Name,
(SELECT 
    COUNT(DISTINCT CONTINENTS.NAME) 
    FROM CONTINENTS
    JOIN PLACES, ENCOUNTERS 
    ON PLACES.ROWI = ENCOUNTERS.PLACE 
    AND ENCOUNTERS.BEAST = BEASTS.ROWI
    AND CONTINENTS.ROWI = PLACES.CONT
) AS Count
FROM BEASTS
WHERE Count > 1
ORDER BY BEASTS.NAME;
